df = pd.DataFrame(papers)
df


model = SentenceTransformer("all-MiniLM-L6-v2")

texts = df["title"] + ". " + df["text"]
embeddings = model.encode(texts.tolist())

dimension = embeddings.shape[1]
index = faiss.IndexFlatL2(dimension)
index.add(np.array(embeddings).astype("float32"))

print("RAG index is ready. Number of papers:", index.ntotal)


def rag_search(query, top_k=3):
    query_embedding = model.encode([query])
    distances, indices = index.search(np.array(query_embedding).astype("float32"), top_k)

    results = []
    for idx in indices[0]:
        paper = df.iloc[idx]
        results.append({
            "title": paper["title"],
            "year": paper["year"],
            "summary": paper["text"]
        })
    return results


question = input("Enter your ecological research question: ")

results = rag_search(question, top_k=3)

print("\nMost relevant academic papers:\n")
for i, r in enumerate(results, 1):
    print(f"{i}. {r['title']} ({r['year']})")
    print(r["summary"])
    print("-" * 80)
